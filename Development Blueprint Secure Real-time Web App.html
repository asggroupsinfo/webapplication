import mongoose from 'mongoose';
import bcrypt from 'bcryptjs';
﻿
const userSchema = new mongoose.Schema(
  {
    name: {
      type: String,
      required: true,
      trim: true,
    },
    email: {
      type: String,
      required: true,
      unique: true,
      lowercase: true,
      trim: true,
    },
    password: {
      type: String,
      required: true,
      select: false, // API call mein default roop se password nahi bheja jayega
    },
    role: {
      type: String,
      enum: ['Super Admin', 'Admin', 'Editor'],
      default: 'Admin',
    },
    status: {
        type: String,
        enum: ['active', 'inactive'],
        default: 'active',
    },
  },
  {
    timestamps: true, // createdAt aur updatedAt fields ko automatically add karta hai
  }
);
﻿
// Password ko save karne se pehle hash karein
userSchema.pre('save', async function (next) {
  if (!this.isModified('password')) return next();
  const salt = await bcrypt.genSalt(10);
  this.password = await bcrypt.hash(this.password, salt);
  next();
});
﻿
const User = mongoose.model('User', userSchema);
export default User;

5. Security aur Authentication Flow (JWT)

Application ko secure karne ke liye JWT (JSON Web Token) based authentication flow ka use kiya jayega.
Flow ke Steps:
  1. Credential Submission:
  • Admin login page par apna email aur password enter karta hai aur "Login" par click karta hai.
  1. API Request:
  • Frontend ( component) POST /api/auth/login endpoint par ek request bhejta hai, jismein request body mein email aur password shamil hota hai.
  1. Server-side Validation:
  • Backend server request receive karta hai.
  • Yeh database mein email ke basis par user ko search karta hai.
  • Agar user mil jaata hai, to yeh bcrypt.compare() ka use karke submit kiye gaye password ko database mein stored hashed password se compare karta hai.
  1. Token Issue karna:
  • Agar credentials sahi hain, to server ek JWT create karta hai. Is token ke payload mein user ko identify karne wali info (jaise userId aur role) aur ek expiry time (exp) shamil hota hai.
  • Token ko .env file mein stored ek secret key (JWT_SECRET) ka use karke digitally sign kiya jaata hai.
  1. Token Bhejna:
  • Server ek success response (200 OK) ke saath JWT ko client ko wapas bhejta hai.
  1. Client-side Storage:
  • Frontend application JWT receive karta hai aur use client-side par store karta hai (for example, localStorage mein).
  1. Secure Requests:
  • Jab bhi client ko kisi secure resource (jaise /api/users) ko access karna hota hai, to woh har API request ke Authorization header mein JWT ko shamil karta hai.
  • Header ka format Authorization: Bearer hoga.
  1. Token Verification (Middleware):
  • Server par, ek auth middleware har secure request ko intercept karta hai.
  • Yeh Authorization header se token nikalta hai, aur JWT_SECRET ka use karke uski validity aur signature ko check karta hai.
  • Agar token valid hai, to middleware payload se user ki info nikal kar request object (e.g., req.user) mein add kar deta hai aur request ko aage badha deta hai.
  • Agar token invalid, missing ya expired hai, to server 401 Unauthorized error return karta hai.
  1. Logout:
  • Logout karne ke liye, frontend localStorage se stored JWT ko हटा deta hai. Iske baad, user ko secure pages access karne se rok diya jaata hai aur login page par redirect kar diya jaata hai.
